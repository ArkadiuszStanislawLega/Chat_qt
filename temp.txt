CREATE TABLE QUESTIONS(ID INTEGER PRIMARY KEY AUTOINCREMENT, VALUE TEXT NOT NULL, ANSWER TEXT NOT NULL);

CREATE TABLE TAGS(ID INTEGER PRIMARY KEY AUTOINCREMENT, TAG TEXT NOT NULL);

CREATE TABLE QUESTIONS_TAGS(ID INTEGER PRIMARY KEY AUTOINCREMENT, 
				QUESTION_ID INTEGER NOT NULL, 
				TAG_ID INTEGER NOT NULL, 
				FOREIGN KEY (QUESTION_ID) REFERENCES QUESTIONS(ID), 
				FOREIGN KEY (TAG_ID) REFERENCES TAGS(ID));
